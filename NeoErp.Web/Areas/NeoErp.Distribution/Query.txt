CREATE OR REPLACE TRIGGER "DIST_RESELLERREGISTRATION_TRG" 
BEFORE INSERT ON DIST_RESELLERREGISTRATION 
FOR EACH ROW
BEGIN
  <<COLUMN_SEQUENCES>>
  BEGIN
    IF INSERTING AND :NEW.ID IS NULL THEN
      SELECT DIST_RESELLERREGISTRATION_SEQ1.NEXTVAL INTO :NEW.ID FROM SYS.DUAL;
    END IF;
  END COLUMN_SEQUENCES;
END;
/
