BATCH TRANSACTION OPENING:
==========================
UPDATE BATCH_TRANSACTION SET LOCATION_CODE =  (SELECT DISTINCT TO_LOCATION_CODE FROM IP_STOCK_LEDGER
WHERE  COMPANY_CODE = BATCH_TRANSACTION.COMPANY_CODE
          AND BRANCH_CODE = BATCH_TRANSACTION.BRANCH_CODE
          AND FORM_CODE = BATCH_TRANSACTION.FORM_CODE
          AND VOUCHER_NO = BATCH_TRANSACTION.REFERENCE_NO
          AND ITEM_CODE = BATCH_TRANSACTION.ITEM_CODE
          AND SERIAL_NO = BATCH_TRANSACTION.SERIAL_NO
          AND FORM_CODE = '0'
          AND VOUCHER_NO = '0' AND ROWNUM = 1
)
WHERE            FORM_CODE = '0'
          AND REFERENCE_NO = '0'
